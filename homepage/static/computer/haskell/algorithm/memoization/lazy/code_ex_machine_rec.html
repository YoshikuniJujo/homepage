type Result v = Maybe (v, String)
run :: String -> Result Int
run s0 = msum [ do
	(c, s) <- char s0
	guard $ isDigit c
	(n, s') <- run s
	('+', s'') <- char s'
	return (n + fromDigit c, s''), do
	(c, s) <- char s0
	guard $ isDigit c
	(n, s') <- run s
	('-', s'') <- char s'
	return (n - fromDigit c, s''),
	return (0, s0) ]

char :: String -> Result Char
char (x : xs) = Just (x, xs)
char _ = Nothing
fromDigit :: Char -> Int
fromDigit '0' = 0
fromDigit '1' = 1
...
fromDigit '9' = 9
fromDigit c = error $ show c ++ " is not digit."

関数runの説明

msum [...]は1番初めに出てきたJust ...を返すものと考えて良い。 次々に試していき初めに成功した値を返すイメージ。

msumの中身はまずは'+'を試し、次に'-'を試し、どちらも一致がなければ、 文字列を消費せずに0を返すといった処理になっている。

'+'を試す部分は以下のようになる。 まずは1文字目が数字かどうかチェック(isDigit c)している。 関数runを再帰的に残りの文字列に適用する。 その結果の残りの文字の1文字目が'+'であること('+', s')を確認し、 結果nにはじめにチェックしたcを数値にしたものを足したものを結果として返している。

'-'を試す部分もほぼ同じことを行っている。
